import QtQuick 2.2 
import QtQuick.Control 1.2 

ApplicationWindow{
    visible: true
    width: 640
    height: 480
    title: qsTr("hello world")
    
    menuBar: MenuBar {
        Menu{
            title: qsTr("File")
            MenuItem{
                text: qsTr("Exit")
                onTriggered: Qt.quit();
            }
        }
    }
    Text {
        text: qsTr("Hello World")
        anchors.centerIn:parent
    }
}
